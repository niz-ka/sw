/*
*  LCM1602 & Arduino Uno
*  VCC - > 5 V
*  GND - GND
*  SCL -> A5
*  SDA -> A4
*/

#include <Wire.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27, 2, 1, 0, 4, 5, 6, 7, 3, POSITIVE);

lcd.begin(16, 2);
lcd.backlight();
lcd.setCursor(0, 0);
lcd.print("Hello, world!");
--------------------------------------------------------------------
/*
*  Speaker
*/

#define SPEAKER A3

pinMode(SPEKAER, OUTPUT);
tone(SPEAKER, 500, 100);

-------------------------------------------------------------------

/*
*  Buttons
*/

#define JUMP_BUTTON 2
#define CROUCH_BUTTON 4

Serial.begin(9600);
pinMode(JUMP_BUTTON, INPUT_PULLUP);
pinMode(CROUCH_BUTTON, INPUT_PULLUP);

if(digitalRead(JUMP_BUTTON) == LOW) {
  Serial.println("Skok!"); 
}
  
if(digitalRead(CROUCH_BUTTON) == LOW) {
  Serial.println("Kucanie!");
}
